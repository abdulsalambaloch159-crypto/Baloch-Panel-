const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); const multiText = document.getElementById('multiplier'); const betBtn = document.getElementById('betBtn'); const cashoutBtn = document.getElementById('cashoutBtn'); let multiplier = 1.00; let isFlying = false; let crashPoint = 0; function startFlight() { isFlying = true; multiplier = 1.00; // کریش پوائنٹ کا فارمولا (مثلاً 1 سے 10 کے درمیان) crashPoint = (Math.random() * 5 + 1).toFixed(2); betBtn.disabled = true; cashoutBtn.disabled = false; let interval = setInterval(() => { if (multiplier >= crashPoint) { clearInterval(interval); isFlying = false; multiText.innerText = "FLEW AWAY!"; multiText.style.color = "red"; betBtn.disabled = false; cashoutBtn.disabled = true; } else { multiplier += 0.01; multiText.innerText = multiplier.toFixed(2) + "x"; multiText.style.color = "white"; } }, 100); // اسپیڈ کنٹرول کرنے کے لیے } betBtn.addEventListener('click', startFlight);
